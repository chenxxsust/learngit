			$(this).val($(this).val().toUpperCase());	
		});

				var flightValue = $("#flight").val();

					var value1 = value.substring(0, 1);
					if (value1 == '0') {
						alert("4位数的航班号中第一个数字为‘0’的话，\n将被‘0’去掉，输入后面3位的数字。\n(例：: CA0623->CA623)");

					}
				}
			}
		});

			"click" : function(e) {
				if ($("#flight").val() == '' || $("#flight1").val() == '') {
					alert($.ShillaCheckout.getMessage("PA016"));
					return false;
				} else {
					if(!flightMap.containsKey($("#flight").val())) {
						alert("航空公司代码无效");
						return false;
					}					
				}
				
				if ($("#flight1").val().length < 3) {
					alert($.ShillaCheckout.getMessage("PA026"));
					return false;
				}
				
				var settingValue = $("#flight").val() + $("#flight1").val();

				
				var airLine = settingValue.substring(0,2);
				var flightName = settingValue.substring(2);					
				
				var resultObj = $.ShillaCheckout.checkDomesticFlight(airLine,flightName);
				if(resultObj.isDomesticFlight) {
					alert(resultObj.message);
					$("#flight1").val("");
					return false;
				}
				
				$("input:hidden[name=flightNo]").val(settingValue);
				$("#airport1").val(settingValue);
				
				// 경유 편명 체크
				var departureDate = $("input:hidden[name=depDate]").val();			
				validateNonStop(departureDate, settingValue);

					if(data) {

							alert($.ShillaCheckout.getMessage("PA109"));

							return false;
						}
						else {
							if('success' == data.result) {
								$("input:hidden[name=deliveryPlaceCode]").val(data.deliveryPlaceCode);
							}
							
							if (confirm($.ShillaCheckout.getMessage("PA025"))) {
								
								popClose($("#layer-searchflight"));					
							}

function isValidExpiredDate(dateStr){
	var date = new Date(dateStr);
	var afterDate = new Date();
	afterDate.setMonth(afterDate.getMonth() + 1);
	afterDate.setDate(afterDate.getDate());
	
	if(date > afterDate){
		return false;
	}
	return true;
	
}

function isValidDate(dateStr) { 
	var date  = new Date(dateStr);
    var KST = 9;
    date.setTime(date.getTime() + date.getTimezoneOffset() * 60000 + (KST * 3600000));
	if(dateStr != date.format('yyyy-MM-dd')){
		return false;		
	}
	return true;
}

function checkPassportRuleNew()
{
	var passNum = $.trim($("#passportNumber_new").val());
	var expireDate = $('#expireDate_year_new').val() + "-" + $('#expireDate_month_new').val() + "-" + $('#expireDate_day_new').val(); 
	
	if(passNum.length == 0){
		alert("请输入护照号码");
		$("#passportNumber_new").focus();
		return false;
	}
	
	//블링크 체크
	if(passNum != $("#passportNumber_new").val()){
		alert("您的护照号码模式有错误。请您再确认您的护照号码。");// 여권번호가 유효하지 않습니다.
		$("#passportNumber_new").focus();
		return false;
	}
	
	
	if(!validateAllPsprtNo(passNum, $("#nationalityCodeNew").val())){
		return false;
	}
	
	
	if(!isValidDate(expireDate) ||  expireDate.length <= 9){
		alert("请输入您的护照有效日期");
		return false;
	}else{
		var passExpireDateTime = $.parseToDate(expireDate);
		var passExpire = new Date(passExpireDateTime.getFullYear(), passExpireDateTime.getMonth(), passExpireDateTime.getDate());
		todayDateTime = new Date();
		today = new Date(todayDateTime.getFullYear(), todayDateTime.getMonth(), todayDateTime.getDate());
		if(today.getTime() > passExpire.getTime()) {
			alert("您的护照有效期不正确，请您再确认一下。");
			return false;
		}
		
		
		todayAfter30 = new Date(todayDateTime.getFullYear() + 30, todayDateTime.getMonth(), todayDateTime.getDate());
		if(todayAfter30.getTime() < passExpire.getTime()) {
			alert("您的护照有效期不正确，请您再确认一下。");
			return false;
		}
		$('#passportExpireDateNew').val(passExpireDateTime);
	}
	
	
	var engLastName = $.trim($("#lastName_new").val());
	var engFirstName = $.trim($("#firstName_new").val());
	
	if((engLastName.length > 14) || (engLastName.length == 0) || (engFirstName.length > 15) || (engFirstName.length == 0)){
		alert("请输入英文名");
		return false;
	}
	if(!engLastName.isUpperCase())
	{
		alert('请以英文大写字母输入您的英文姓名。'); // 영문성명은 영문 대문자로 입력해주시기 바랍니다.
		$('#lastName_new').focus();
		return false;
	}
	if(!engFirstName.isUpperCase())
	{
		alert('请以英文大写字母输入您的英文姓名。'); // 영문성명은 영문 대문자로 입력해주시기 바랍니다.
		$('#firstName_new').focus();
		return false;
	}
	
	return true;
}


function validateAllPsprtNo(psprtNo,nationCode) {

     	},
     	async: false,
     	error: function(){
			alert("故障发生在与服务器的通信。");
     	},
     	success: function(resultMap) {
     		if(resultMap.result == "success") {

     		} else {
     			alert(resultMap.errorMsg);

}

function isValidDate(dateStr) { 
	var date  = new Date(dateStr);
    var KST = 9;
    date.setTime(date.getTime() + date.getTimezoneOffset() * 60000 + (KST * 3600000));
	if(dateStr != date.format('yyyy-MM-dd')){
		return false;		
	}
	return true;
}

$.parseToDate = function(sourceDate) {
	   var yyyy  = sourceDate.substring(0,4);
	   var month = sourceDate.substring(5,7);
	   var dd    = sourceDate.substring(8,10);
	   
	   var targetDate = new Date(yyyy, month - 1, dd);
	   
	   return targetDate;
};
